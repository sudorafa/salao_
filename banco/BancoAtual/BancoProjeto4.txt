CREATE TABLE Usuario (
IdUsuario Int PRIMARY KEY,
tipoUsuario Int,
Login nvarchar(100) unique,
Senha nvarchar(100),
Nome nvarchar(20)
)

CREATE TABLE Cliente (
CPF nvarchar(20) PRIMARY KEY,
Endereço nvarchar(200),
Telefone nvarchar(30),
DataNascimento nvarchar(20),
Email nvarchar(20)
)

CREATE TABLE Funcionario (
CPF nvarchar(20) PRIMARY KEY,
Nome nvarchar(100),
Telefone nvarchar(30),
Endereco nvarchar(200),
Porcentagem float,
DataNascimento nvarchar(20)
)


CREATE TABLE Serviços (
IdServico int PRIMARY KEY,
Descricao nvarchar(100),
Valor money
)

CREATE TABLE Produto (
IdProduto int PRIMARY KEY,
Descricao nvarchar(20),
Valor Money,
Quantidade Int
)

CREATE TABLE Venda (
IdRelatorio Int PRIMARY KEY,
ValorTotal money,
Data nvarchar(20),
IdServico int,
IdProduto int,
CPFFuncionario nvarchar(20),
CPFCliente nvarchar(20),
FOREIGN KEY(IdServico) REFERENCES Serviço (IdServico),
FOREIGN KEY(IdProduto) REFERENCES Produtos (IdProduto),
FOREIGN KEY(CPFFuncionario) REFERENCES Funcionario (CPF),
FOREIGN KEY(CPFCliente) REFERENCES Cliente (CPF)
)
